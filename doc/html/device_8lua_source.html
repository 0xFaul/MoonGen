254 --- Set the tx rate of a queue in MBit/s.
255 -- This sets the payload rate, not to the actual wire rate, i.e. preamble, SFD, and IFG are ignored.
256 -- The X540 and 82599 chips seem to have a hardware bug (?): they seem use the wire rate in some point of the throttling process.
257 -- This causes erratic behavior for rates >= 64/84 * WireRate when using small packets.
258 -- The function is non-linear (not even monotonic) for such rates.
259 -- The function prints a warning if such a rate is configured.
260 -- A simple work-around for this is using two queues with 50% of the desired rate.
261 -- Note that this changes the inter-arrival times as the rate control of both queues is independent.
262 function txQueue:setRate(rate)
263  if self.dev:getPciId() ~= mod.PCI_ID_82599 and self.dev:getPciId() ~= mod.PCI_ID_X540 then
264  error("tx rate control not yet implemented for this NIC")
265  end
266  local speed = self.dev:getLinkStatus().speed
267  if speed <= 0 then
268  print("WARNING: link down, assuming 10 GbE connection")
269  speed = 10000
270  end
271  if rate <= 0 then
272  rate = speed
273  end
274  self.rate = math.min(rate, speed)
275  self.speed = speed
276  local link = self.dev:getLinkStatus()
277  self.speed = link.speed
278  rate = rate / speed
279  -- the X540 and 82599 chips have a hardware bug: they assume that the wire size of an
280  -- ethernet frame is 64 byte when it is actually 84 byte (8 byte preamble/SFD, 12 byte IFG)
281  -- TODO: software fallback for bugged rates and unsupported NICs
282  if rate >= (64 * 64) / (84 * 84) and rate < 1 then
283  print("WARNING: rates with a payload rate >= 64/84% do not work properly with small packets due to a hardware bug, see documentation for details")
284  end
285  if rate <= 0 then
286  error("rate must be > 0")
287  end
288  if rate >= 1 then
289  self:setTxRateRaw(0, true)
290  else
291  self:setTxRateRaw(1 / rate)
292  end
293 end

303 function txQueue:setTxRateRaw(rate, disable)
304  dpdkc.write_reg32(self.id, RTTDQSEL, self.qid)
305  if disable then
306  dpdkc.write_reg32(self.id, RF_X540_82599, 0)
307  return
308  end
309  -- 10.14 fixed-point
310  local rateInt = math.floor(rate)
311  local rateDec = math.floor((rate - rateInt) * 2^14)
312  dpdkc.write_reg32(self.id, RF_X540_82599, bit.bor(bit.lshift(rateInt, 14), rateDec, RF_ENABLE_BIT))
313 end

315 function txQueue:getTxRate()
316  local link = self.dev:getLinkStatus()
317  self.speed = link.speed > 0 and link.speed or 10000
318  dpdkc.write_reg32(self.id, RTTDQSEL, self.qid)
319  local reg = dpdkc.read_reg32(self.id, RF_X540_82599)
320  if reg == 0 then
321  self.rate = nil
322  return self.speed
323  end
324  -- 10.14 fixed-point
325  local rateInt = bit.band(bit.rshift(reg, 14), 0x3FFF)
326  local rateDec = bit.band(reg, 0x3FF)
327  self.rate = (1 / (rateInt + rateDec / 2^14)) * self.speed
328  return self.rate
329 end
