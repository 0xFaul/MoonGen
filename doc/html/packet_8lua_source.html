30 --- Retrieve the time stamp information.
31 --- @return The timestamp or nil if the packet was not time stamped.
32 function pkt:getTimestamp()
33  if bit.bor(self.ol_flags, dpdk.PKT_RX_IEEE1588_TMST) ~= 0 then
34  -- TODO: support timestamps that are stored in registers instead of the rx buffer
35  local data = ffi.cast("uint32_t* ", self.pkt.data)
36  -- TODO: this is only tested with the Intel 82580 NIC at the moment
37  -- the datasheet claims that low and high are swapped, but this doesn't seem to be the case
38  -- TODO: check other NICs
39  local low = data[2]
40  local high = data[3]
41  return high * 2^32 + low
42  end
43 end

89 --- Instruct the NIC to calculate the IP checksum for this packet.
90 --- @param ipv4 Boolean to decide whether the packet uses IPv4 (set to nil/true) or IPv6 (set to anything else).
91 --- In case it is an IPv6 packet, do nothing (the header has no checksum).
92 --- @param l2_len Length of the layer 2 header in bytes (default 14 bytes for ethernet).
93 --- @param l3_len Length of the layer 3 header in bytes (default 20 bytes for IPv4).
94 function pkt:offloadIPChecksum(ipv4, l2_len, l3_len)
95  -- NOTE: this method cannot be moved to the udpPacket class because it doesn't (and can't) know the pktbuf it belongs to
96  ipv4 = ipv4 == nil or ipv4
97  if ipv4 then
98  l2_len = l2_len or 14
99  l3_len = l3_len or 20
100  self.ol_flags = bit.bor(self.ol_flags, dpdk.PKT_TX_IPV4_CSUM)
101  self.pkt.header_lengths = l2_len * 512 + l3_len
102  end
103 end

109 function pkt:offloadUdpChecksum(ipv4, l2_len, l3_len)
110  -- NOTE: this method cannot be moved to the udpPacket class because it doesn't (and can't) know the pktbuf it belongs to
111  ipv4 = ipv4 == nil or ipv4
112  l2_len = l2_len or 14
113  if ipv4 then
114  l3_len = l3_len or 20
115  self.ol_flags = bit.bor(self.ol_flags, dpdk.PKT_TX_IPV4_CSUM, dpdk.PKT_TX_UDP_CKSUM)
116  self.pkt.header_lengths = l2_len * 512 + l3_len
117  -- calculate pseudo header checksum because the NIC doesn't do this...
118  dpdkc.calc_ipv4_pseudo_header_checksum(self.pkt.data, 20)
119  else

122  self.pkt.header_lengths = l2_len * 512 + l3_len

124  dpdkc.calc_ipv6_pseudo_header_checksum(self.pkt.data, 30)

275 --- Set all members of all headers.
276 --- Per default, all members are set to default values specified in the respective set function.
277 --- Optional named arguments can be used to set a member to a user-provided value.
278 --- The argument 'pktLength' can be used to automatically calculate and set the length member of headers (e.g. ip header).
279 --- @code
280 --- fill() --- only default values
281 --- fill{ ethSrc="12:23:34:45:56:67", ipTTL=100 } --- all members are set to default values with the exception of ethSrc and ipTTL
282 --- fill{ pktLength=64 } --- only default values, length members of the headers are adjusted
283 --- @endcode
284 --- @param self The packet
285 --- @param args Table of named arguments. For a list of available arguments see "See also"
286 --- @note This function is slow. If you want to modify members of a header during a time critical section of your script use the respective setters.
287 function packetFill(self, namedArgs)
288  -- fill headers
289  local headers = self:getHeaders()
290  local args = self:getArgs()
291  local accumulatedLength = 0
292  for i, v in ipairs(headers) do
293  local curMember = args[i]
294  if type(curMember) == "table" then
295  curMember = curMember[2]
296  end
297  local nextHeader = args[i + 1]
298  if type(nextHeader) == "table" then
299  nextHeader = nextHeader[1]
300  end
301  namedArgs = v:setDefaultNamedArgs(curMember, namedArgs, nextHeader, accumulatedLength)
302  v:fill(namedArgs, curMember)
303 
304  accumulatedLength = accumulatedLength + ffi.sizeof(v)
305  end
306 end

328 function packetResolveLastHeader(self)
329  local name = self:getName()
330  local headers = self:getHeaders()
331  local next_header = headers[#headers]:resolveNextHeader()
332

395 --- Creates a packet struct (cdata) consisting of different headers.
396 --- Simply list the headers in the order you want them to be in a packet.
397 --- If you want the member to be named differently, use the following syntax:
398 --- normal: <header> ; different membername: { <header>, <member> }.
399 --- Supported keywords: eth, arp, ptp, ip, ip6, udp, tcp, icmp
400 --- @code
401 --- makeStruct('eth', { 'ip4', 'ip' }, 'udp') --- creates an UDP packet struct
402 --- --- the ip4 member of the packet is named 'ip'
403 --- @endcode
404 --- @param args list of keywords/tables of keyword-member pairs
405 --- @return name name of the struct
406 --- @return ctype ctype of the struct
407 function packetMakeStruct(...)
408  local name = ""
409  local str = ""
410 
411  -- add the specified headers and build the name
412  for _, v in ipairs(...) do
413  local header, member
414  if type(v) == "table" then
415  header = v[1]
416  member = v[2]
417  else
418  header = v
419  member = v
420  end
421 
422  -- alias for eth -> ethernet
423  if header == 'eth' then
424  header = 'ethernet'
425  end
426 
427  -- add header
428  str = str .. [[
429  struct ]] .. header .. '_header ' .. member .. [[;
430  ]]
431 
432  -- build name
433  name = name .. "__" .. header .. "_" .. member
434  end
435 
436  -- add rest of the struct
437  str = [[
438  struct __attribute__((__packed__)) ]]
439  .. name
440  .. [[ {
441  ]]
442  .. str
443  .. [[
444  union payload_t payload;
445  };
446  ]]
447  -- add struct definition, return full name and typeof
448  ffi.cdef(str)
449  name = "struct " .. name
450  return name, ffi.typeof(name .. "*")
451 end
