171 -- Per default, all members are set to default values specified in the respective set function.
172 -- Optional named arguments can be used to set a member to a user-provided value.
173 -- @param args Table of named arguments. Available arguments: ethSrc, ethDst, ethType
174 -- @param pre prefix for namedArgs. Default 'eth'.

176 -- @usage fill{ ethSrc="12:23:34:45:56:67", ethType=0x137 } -- default value for ethDst; ethSrc and ethType user-specified
177 function etherHeader:fill(args, pre)
178  args = args or {}
179  pre = pre or "eth"
180 
181  local src = pre .. "Src"
182  local dst = pre .. "Dst"
183  args[src] = args[src] or "01:02:03:04:05:06"
184  args[dst] = args[dst] or "07:08:09:0a:0b:0c"
185 
186  -- addresses can be either a string, a mac_address ctype or a device/queue object
187  if type(args[src]) == "string" then
188  self:setSrcString(args[src])
189  elseif istype(macAddrType, args[src]) then
190  self:setSrc(args[src])
191  elseif type(args[src]) == "table" and args[src].id then
192  self:setSrcString((args[src].dev or args[src]):getMacString())
193  end
194  if type(args[dst]) == "string" then
