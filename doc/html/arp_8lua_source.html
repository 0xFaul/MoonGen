319 local function arpTask(rxQueue, txQueue, ips)
320  arpTable.taskRunning = true
321  if type(ips) ~= "table" then
322  ips = { ips }
323  end
324  local ipToMac = {}
325  for i, v in ipairs(ips) do
326  if type(v) == "string" then
327  v = parseIPAddress(v)
328  ips[i] = v
329  end
330  ipToMac[v] = true -- TODO: support different MACs for different IPs
331  end
332  if rxQueue.dev ~= txQueue.dev then
333  error("both queues must belong to the same device")
334  end
335  local arpSrcIP = ips[1] -- the source address for ARP requests
336 
337  local dev = rxQueue.dev
338  local devMac = dev:getMac()
339  local rxBufs = memory.createBufArray(1)
340  local txMem = memory.createMemPool(function(buf)
341  buf:getArpPacket():fill{
342  ethSrc = devMac,
343  arpOperation = arp.OP_REPLY,
344  arpHardwareSrc = devMac,
345  arpProtoSrc = devIP,
346  pktLength = 60
347  }
348  end)
349  local txBufs = txMem:bufArray(1)
350  dev:l2Filter(eth.TYPE_ARP, rxQueue)
351 
352  while dpdk.running() do
353  rx = rxQueue:tryRecvIdle(rxBufs, 1000)
354  assert(rx <= 1)
355  if rx > 0 then
356  local rxPkt = rxBufs[1]:getArpPacket()
357  if rxPkt.eth:getType() == eth.TYPE_ARP then
358  if rxPkt.arp:getOperation() == arp.OP_REQUEST then
359  local ip = rxPkt.arp:getProtoDst()
360  local mac = ipToMac[ip]
361  if mac then
362  if mac == true then
363  mac = devMac
364  end
365  txBufs:alloc(60)
366  -- TODO: a single-packet API would be nice for things like this
367  local pkt = txBufs[1]:getArpPacket()
368  pkt.eth:setDst(rxPkt.eth:getSrc())
369  pkt.arp:setOperation(arp.OP_REPLY)
370  pkt.arp:setHardwareDst(rxPkt.arp:getHardwareSrc())
371  pkt.arp:setProtoDst(rxPkt.arp:getProtoSrc())
372  pkt.arp:setProtoSrc(ip)
373  txQueue:send(txBufs)
374  end
375  elseif rxPkt.arp:getOperation() == arp.OP_REPLY then
376  -- learn from all arp replies we see (suspicable to arp cache poisoning but that doesn't matter here)
377  local mac = rxPkt.arp:getHardwareSrcString()
378  local ip = rxPkt.arp:getProtoSrcString()
379  arpTable[tostring(parseIPAddress(ip))] = { mac = mac, timestamp = dpdk.getTime() }
380  end
381  end
382  rxBufs:freeAll()
383  end
384  -- send outstanding requests
385  arpTable:forEach(function(ip, value)
386  -- TODO: refresh or GC old entries
387  if value ~= "pending" then
388  return
389  end
390  arpTable[ip] = "requested"
391  -- TODO: the format should be compatible with parseIPAddress
392  ip = tonumber(ip)
393  txBufs:alloc(60)
394  local pkt = txBufs[1]:getArpPacket()
395  pkt.eth:setDstString(eth.BROADCAST)
396  pkt.arp:setOperation(arp.OP_REQUEST)
397  pkt.arp:setHardwareDstString(eth.BROADCAST)
398  pkt.arp:setProtoDst(ip)
399  pkt.arp:setProtoSrc(arpSrcIP)
400  txQueue:send(txBufs)
401  end)
402  --dpdk.sleepMillisIdle(1)
403  end
404 end
